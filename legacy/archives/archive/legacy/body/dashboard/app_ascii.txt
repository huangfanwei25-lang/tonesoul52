            history = st.session_state.spine.history
            
            # Extract Coordinates
            x_vals = [h.get('S', {}).get('score', 0.5) if isinstance(h.get('S'), dict) else h.get('S', 0.5) for h in history]
            y_vals = [h.get('T', {}).get('score', 0.5) if isinstance(h.get('T'), dict) else h.get('T', 0.5) for h in history]
            z_vals = [h.get('I', {}).get('score', 0.5) if isinstance(h.get('I'), dict) else h.get('I', 0.5) for h in history]
            
            # Extract Labels from Chat History
            # We map 1:1 if possible, or use indexing.
            # Chat History has USER and ASSISTANT messages. 
            # Spine History tracks states AFTER processing User Input.
            # So index i in Spine History roughly corresponds to User Message i (if we filter just users).
            
            user_msgs = [m['content'] for m in st.session_state.chat_history if m['role'] == 'user']
            # Align lengths
            min_len = min(len(x_vals), len(user_msgs))
            labels = [user_msgs[i][:20] + "..." for i in range(min_len)]
            
            # If spine history is longer (e.g. from previous sessions?), pad with "State N"
            if len(x_vals) > len(labels):
                labels += [f"State {i}" for i in range(len(labels), len(x_vals))]

            # 1. Trajectory Line (The Thought Chain)

# --- TAB 1: CONSCIOUS (Chat) ---
with tab1:
    st.subheader("The Echo Interface")
    st.caption("Governance Loop: Input -> Telemetry -> Guardian -> LLM -> Ledger -> Output")
    
    # Display Chat
    for msg in st.session_state.chat_history:
        if msg['role'] == 'user':
            st.chat_message("user").write(msg['content'])
        else:
            st.chat_message("assistant").write(msg['content'])

    # Input
    col_input, col_file = st.columns([0.85, 0.15])
    with col_input:
        prompt = st.chat_input("Speak to the governed machine...")
    with col_file:
        uploaded_file = st.file_uploader("???", type=["jpg", "png", "jpeg"], label_visibility="collapsed")

    if prompt:
        # 1. User Message
        st.session_state.chat_history.append({"role": "user", "content": prompt})
        
        # Handle Image
        image_path = None
        if uploaded_file:
            # Display Image in Chat
            st.chat_message("user").image(uploaded_file, caption="Visual Input", use_column_width=True)
            # Save Temp
            os.makedirs("temp/uploads", exist_ok=True)
            image_path = os.path.join("temp/uploads", uploaded_file.name)
            with open(image_path, "wb") as f:
                f.write(uploaded_file.getbuffer())
        
        st.chat_message("user").write(prompt)
        
        # 2. Process via Spine (The Core Loop)
        with st.spinner("Auditing Input & Generating Response (Vision Active)..."):
            response_text = st.session_state.spine.process_input(prompt, image_path=image_path)
            
            # Update Metrics (Vision-Aware)
            try:
                # If image exists, metrics are already updated inside process_input,
                # but we can grab the latest from history if available.
                if hasattr(st.session_state.spine, 'history') and st.session_state.spine.history:
                    latest = st.session_state.spine.history[-1]
                    # Handle flat or nested dict
                    st.session_state.last_metrics = {
                        'S': latest.get('S', {}).get('score', latest.get('S', 0.5)),
                        'T': latest.get('T', {}).get('score', latest.get('T', 0.5)),
                        'R': latest.get('R', {}).get('score', latest.get('R', 0.5)),
                        'E': latest.get('E', {}).get('score', latest.get('E', 0.5)),
                        'I': latest.get('I', {}).get('score', latest.get('I', 0.5))
                    }
            except Exception as e:
                print(f"Metrics Update Error: {e}")
        
        # 3. Assistant Message
        st.session_state.chat_history.append({"role": "assistant", "content": response_text})
        st.chat_message("assistant").write(response_text)
        st.rerun()

    # 4. Emotional Trend (Line Chart)
    if hasattr(st.session_state.spine, 'history') and st.session_state.spine.history:
        st.divider()
        st.subheader("Emotional Arc (Trend)")
        st.caption("The Soul's changing state over time.")
        
        # Convert history list of dicts to DataFrame
        # Data structure: [{'S': {'score': 0.5, 'grade': 'MID'}, ...}, ...]
        # We need to flatten this to plot scores.
        flattened_history = []
        for entry in st.session_state.spine.history:
            flat_entry = {}
            for k, v in entry.items():
                if isinstance(v, dict) and 'score' in v:
                    flat_entry[k] = v['score']
                else:
                    flat_entry[k] = v # Fallback
            flattened_history.append(flat_entry)
            
        history_df = pd.DataFrame(flattened_history)
        
        # Show all 5 metrics
        st.line_chart(history_df[['S', 'T', 'R', 'E', 'I']])

    # Ledger View (Bottom of Tab 1)
    st.divider()
    with st.expander("Immutable Ledger (Protocol L4)", expanded=True):
        real_ledger = os.path.join(os.path.dirname(__file__), '..', '..', 'body', 'ledger', 'ledger.jsonl')
        if os.path.exists(real_ledger):
            with open(real_ledger, 'r', encoding='utf-8') as f:
                lines = f.readlines()
                data = [json.loads(line) for line in lines[-5:]]
                if data:
                    df = pd.DataFrame(data)
                    st.dataframe(df, use_container_width=True)
        else:
            st.warning("No Ledger Found.")
